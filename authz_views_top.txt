"""
Keycloak OIDC Authorization Code flow for CyberPanel.

Replaces CyberPanel's native login page with Keycloak SSO:
  /auth/login/    GÍó redirects to Keycloak authorization endpoint
  /auth/callback/ GÍó exchanges code for tokens, creates CyberPanel session
  /auth/logout/   GÍó clears session, redirects to Keycloak logout
  /auth/sso/      GÍó direct token-based SSO (for Vue frontend integration)
  / (root)        GÍó if authenticated go to dashboard, else redirect to Keycloak
"""

# ---------------------------------------------------------------------------
# Keycloak OIDC endpoints (derived from settings)
# ---------------------------------------------------------------------------
KEYCLOAK_SERVER_URL = getattr(settings, 'KEYCLOAK_SERVER_URL', 'https://key.hcos.io').rstrip('/')
KEYCLOAK_PUBLIC_URL = getattr(settings, 'KEYCLOAK_PUBLIC_URL', KEYCLOAK_SERVER_URL).rstrip('/')
KEYCLOAK_REALM = getattr(settings, 'KEYCLOAK_REALM_NAME', 'master')
KEYCLOAK_CLIENT_ID = getattr(settings, 'KEYCLOAK_CLIENT_ID', 'hcos-backend')
KEYCLOAK_CLIENT_SECRET = getattr(settings, 'KEYCLOAK_CLIENT_SECRET', '')

# Browser-facing endpoints (user's browser redirects here)
AUTH_ENDPOINT = f"{KEYCLOAK_PUBLIC_URL}/realms/{KEYCLOAK_REALM}/protocol/openid-connect/auth"
LOGOUT_ENDPOINT = f"{KEYCLOAK_PUBLIC_URL}/realms/{KEYCLOAK_REALM}/protocol/openid-connect/logout"
# Server-to-server endpoints (CyberPanel container calls these directly)
TOKEN_ENDPOINT = f"{KEYCLOAK_SERVER_URL}/realms/{KEYCLOAK_REALM}/protocol/openid-connect/token"
USERINFO_ENDPOINT = f"{KEYCLOAK_SERVER_URL}/realms/{KEYCLOAK_REALM}/protocol/openid-connect/userinfo"
JWKS_URL = f"{KEYCLOAK_SERVER_URL}/realms/{KEYCLOAK_REALM}/protocol/openid-connect/certs"

# SSL context for Keycloak HTTP calls
_verify_ssl = getattr(settings, 'KEYCLOAK_VERIFY_SSL', 'true').lower() != 'false'
